Document ID: A7K2 (The unique 4-digit code)   
groupName: "The Smith Family"allowedSeats: 2   
hasRSVPd: true / false
submittedData: (Object containing names, allergies, music requests, and accommodation status)   

firestore security rules:
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Admin access: Allow everything if the user is authenticated as an admin
    match /{document=**} {
      allow read, write: if request.auth != null && request.auth.token.isAdmin == true;
    }

    // Guest access: Only allow reading/writing a specific document if you know its exact 4-digit ID
    match /guests/{guestCode} {
      // Guests can read their own data or update it to submit their RSVP
      allow read, update: if true; 
      
      // Prevent guests from creating new codes or deleting existing ones
      allow create, delete: if false; 
    }
  }
}
